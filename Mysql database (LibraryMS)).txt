CREATE DATABASE library;



CREATE TABLE books(
    BookId VARCHAR(30),
    Title VARCHAR(50) NOT NULL,
     Author VARCHAR(50) NOT NULL,
     Publisher VARCHAR(50) NOT NULL,
     Genre VARCHAR(50) NOT NULL,
     Language VARCHAR(50),
     NumberOfPages INT,
     PRIMARY KEY(BookId)
        );

CREATE TABLE publishers(
       PublisherId VARCHAR(20),
       PublisherName VARCHAR (50) NOT NULL,
       EmailOrAddress VARCHAR (100) NOT NULL,
       ContactNumber VARCHAR(12) UNIQUE,
       PRIMARY KEY (PublisherId)
    );

CREATE TABLE genres(
 GenreId VARCHAR(30) ,
 GenreName VARCHAR(50),
 PRIMARY KEY(GenreId)
 );

CREATE TABLE members(
  UserId VARCHAR(30),
  UserName VARCHAR(50)NOT NULL,
  First_Name VARCHAR(50)NOT NULL,
  Last_Name VARCHAR(50) NOT NULL,
  Email VARCHAR(40) NOT NULL DEFAULT 'No Email',
  ContactNumber VARCHAR(12) UNIQUE,
  Address VARCHAR(255) NOT NULL DEFAULT 'No Address',
  Position VARCHAR(50) NOT NULL,
  Join_Date DATE NOT NULL,
  PRIMARY KEY(UserId)
 );


CREATE TABLE transactions(
      TransactionId VARCHAR(30),
      UserId VARCHAR(30),
      BookId VARCHAR(30),
      IssueDate DATE NOT NULL,
      DueDate DATE NOT NULL,
      PRIMARY KEY(TransactionId),
      FOREIGN KEY(UserId) REFERENCES members(UserId),
      FOREIGN KEY(BookId) REFERENCES Books(BookId)
       );

CREATE TABLE fine(
  FineId INT,
  TransactionId VARCHAR(30),
  DueDate DATE NOT NULL,
  ReturnDate DATE NOT NULL,
  FineAmount DECIMAL(10,2) NOT NULL,
  PaymentDate VARCHAR(100) NOT NULL,
  PRIMARY KEY(FineId),
  FOREIGN KEY(TransactionId) REFERENCES transactions(TransactionId)
);
